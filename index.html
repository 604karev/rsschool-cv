    let isPlay = false;
    let timeOutId;
    const listLength = playList.length - 1;
    const convertToMilliseconds = (min, sec) => (parseInt(min) * 60 + parseInt(sec)) * 1000;

    const getTrackNumber = (playNum = 0) => {
        return (direction = '') => {
            switch (direction) {
                case 'NEXT':
                    let nextDirection = playNum >= listLength ? playNum = 0 : ++playNum;
                    return getTrack(nextDirection);
                case 'PREV':
                    let prevDirection = playNum <= 0 ? playNum = listLength : --playNum;
                    return getTrack(prevDirection);
                default:
                    return getTrack(playNum)
            }
        }
    };

    const playTrack = getTrackNumber();

    const getTrack = (playNum) => {
        clearTimeout(timeOutId);
        const list = document.querySelectorAll('.play-item');
        list.forEach(item => item.classList.remove('item-active'));
        list[playNum].classList.add('item-active');
        audio.src = playList[playNum].src;
        isPlay ? audio.play() : audio.pause();
        const durationArray = playList[playNum].duration.split(':');
        const duration = convertToMilliseconds(...durationArray);
        timeOutId = setTimeout(() => playTrack('NEXT'), duration);
    };
